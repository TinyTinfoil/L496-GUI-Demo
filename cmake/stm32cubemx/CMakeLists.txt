    message(ERROR "Generated code requires C++11 or higher")
endif()

target_sources(dsp PRIVATE ${dsp_Src})
target_link_libraries(dsp PUBLIC stm32cubemx)

target_link_libraries(${CMAKE_PROJECT_NAME} ${MX_LINK_LIBS})
